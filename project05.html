plot_slopegraph <- function(df, title_text) {

  long <- df %>%
    pivot_longer(
      cols = c(Turnout_Rate, Predicted_Turnout),
      names_to = "Type",
      values_to = "Value"
    ) %>%
    mutate(
      Type = factor(Type, levels = c("Predicted_Turnout", "Turnout_Rate")),
      Label = ifelse(Type == "Turnout_Rate", "Actual", "Predicted")
    )

  # Order counties by error magnitude
  order <- df %>%
    mutate(Error = Turnout_Rate - Predicted_Turnout) %>%
    arrange(Error) %>%
    pull(County)

  long$County <- factor(long$County, levels = order)

  # Highlight top 25% largest absolute errors
  long <- long %>%
    left_join(
      df %>% 
        mutate(Error = Turnout_Rate - Predicted_Turnout,
               Highlight = abs(Error) > quantile(abs(Error), 0.75)) %>%
        select(County, Highlight),
      by = "County"
    )

  ggplot(long, aes(Type, Value, group = County)) +
    geom_line(aes(color = Highlight), linewidth = 1) +
    geom_point(aes(color = Highlight), size = 3) +

    # Labels on left (Predicted)
    geom_text_repel(
      data = long %>% filter(Type == "Predicted_Turnout"),
      aes(label = County),
      nudge_x = -0.2,
      size = 3,
      color = text_col,
      segment.color = NA
    ) +

    # Labels on right (Actual)
    geom_text_repel(
      data = long %>% filter(Type == "Turnout_Rate"),
      aes(label = County),
      nudge_x = 0.2,
      size = 3,
      color = text_col,
      segment.color = NA
    ) +

    scale_color_manual(
      values = c("FALSE" = "#bfbfbf", "TRUE" = "#d73a1f"),
      guide = "none"
    ) +
    scale_x_discrete(
      labels = c("Predicted_Turnout" = "Predicted", "Turnout_Rate" = "Actual")
    ) +
    scale_y_continuous(
      breaks = seq(40, 90, 5),
      labels = function(x) sprintf("%.0f", x)
    ) +
    theme_minimal(base_family = "sans") +
    theme(
      axis.title = element_blank(),
      panel.grid.minor = element_blank(),
      panel.grid.major.x = element_blank(),
      plot.title = element_text(face = "bold", size = 16, hjust = 0.5),
      axis.text.x = element_text(face = "bold", size = 12),
      plot.background = element_rect(fill = fill_col, color = NA)
    ) +
    ggtitle(title_text)
}
