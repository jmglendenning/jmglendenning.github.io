# === Step 4: Coerce vote columns to numeric, compute total_votes and turnout ===
vote_cols <- c("G20PRERTRU", "G20PREDBID", "G20PRELJOR", "G20PREGHAW", "G20PREOWRI")

precincts <- precincts %>%
  mutate(across(all_of(vote_cols), ~ as.numeric(as.character(.)))) %>%  # Ensure numeric
  mutate(
    total_votes = rowSums(across(all_of(vote_cols)), na.rm = TRUE),
    participation_rate = total_votes / as.numeric(G20VR)
  ) %>%
  filter(is.finite(participation_rate) & participation_rate > 0)

# === Step 5: Create turnout quintiles ===
precincts <- precincts %>%
  mutate(
    turnout_quintile = cut(
      participation_rate,

      include.lowest = TRUE,
      labels = c("Q1 (Lowest)", "Q2", "Q3", "Q4", "Q5 (Highest)")
    )
  )

hlth <- read.csv("/Users/john/Library/Mobile Documents/com~apple~CloudDocs/Home/John/GIS/EPPS6356/Project/nhgis0025_csv/nhgis0025_ds250_20205_county.csv")

edu <- edu %>%

hlth <- hlth %>%
  filter(STUSAB == "TX") %>%
  mutate(uninsured = ANB0M169 / 100) %>%
  select(GISJOIN, uninsured, COUNTY, COUNTYA)

# === Combine metrics ===
civic_county <- left_join(edu, hlth, by = c("GISJOIN", "COUNTY", "COUNTYA"))

# === Aggregate turnout at county level ===
precincts <- precincts %>%
  mutate(across(c(G20PRERTRU, G20PREDBID, G20PRELJOR, G20PREGHAW, G20PREOWRI), ~ as.numeric(as.character(.)))) %>%
  mutate(
    total_votes = rowSums(across(starts_with("G20PRE")), na.rm = TRUE),
    participation_rate = total_votes / as.numeric(G20VR),
    fips = sprintf("%03s", as.character(CNTY))
  )

county_turnout <- precincts %>%
  st_drop_geometry() %>%
  group_by(fips) %>%
  summarise(avg_turnout = mean(participation_rate, na.rm = TRUE))

civic_county <- civic_county %>%
  mutate(fips = sprintf("%03d", COUNTYA)) %>%
  left_join(county_turnout, by = "fips")

# === Classify counties ===
civic_county <- civic_county %>%
  mutate(
    edu_bin = ifelse(pct_bachelor > median(pct_bachelor, na.rm = TRUE), "High", "Low"),
    hlth_bin = ifelse(uninsured < median(uninsured, na.rm = TRUE), "High", "Low"),
    turnout_bin = ifelse(avg_turnout > median(avg_turnout, na.rm = TRUE), "High", "Low"),
    civic_type = case_when(
      edu_bin == "High" & hlth_bin == "High" & turnout_bin == "High" ~ "Civic Garden",
      edu_bin == "Low"  & hlth_bin == "Low"  & turnout_bin == "Low"  ~ "Civic Desert",
      TRUE ~ "Mixed"
    )
  )

  left_join(civic_county %>% select(fips, civic_type), by = "fips")

# === Prepare analysis dataset ===
anova_data <- civic_county %>%
  mutate(
    civic_type = case_when(
      pct_bachelor > median(pct_bachelor, na.rm = TRUE) &
      uninsured < median(uninsured, na.rm = TRUE) &
      avg_turnout > median(avg_turnout, na.rm = TRUE) ~ "Civic Garden",
      
      pct_bachelor <= median(pct_bachelor, na.rm = TRUE) &
      uninsured >= median(uninsured, na.rm = TRUE) &
      avg_turnout <= median(avg_turnout, na.rm = TRUE) ~ "Civic Desert",
      
      TRUE ~ "Mixed"
    )
  ) %>%
  filter(
    is.finite(pct_bachelor),
    is.finite(uninsured),
    is.finite(avg_turnout)
  ) %>%
  select(civic_type, pct_bachelor, uninsured, avg_turnout) %>%
  filter(complete.cases(.))  # drop NAs
