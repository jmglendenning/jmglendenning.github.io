def extract_clean_paragraphs_from_pdf(pdf_path,
                                      min_char_per_paragraph=50,
                                      font_size_threshold=7,
                                      split_large_paragraphs=True,
                                      max_words_per_chunk=150,
                                      exclusion_path="data/exclusion.txt",
                                      regex_patterns=None):

    # Load dependencies once
    exclusion_words = set(open(exclusion_path).read().lower().split())
    symspell = load_symspell(symspell_dict)  # Adjust path as needed
    nlp = spacy.load("en_core_web_sm")
    
    # Default patterns
    if regex_patterns is None:
        regex_patterns = [
            r'\[\d+\]', r'BIBLIOGRAPHY[\s\S]*',
            r'This content downloaded from[\s\S]*?[-]{5,}'
        ]

    # Shared mutable context
    context = {
        "paragraphs": [],
        "current_paragraph": "",
        "header_footer_texts": [],
        "min_char_per_paragraph": min_char_per_paragraph,
        "font_size_threshold": font_size_threshold,
        "split_large_paragraphs": split_large_paragraphs,
        "max_words_per_chunk": max_words_per_chunk,
        "regex_patterns": regex_patterns,
        "exclusion_words": exclusion_words,
        "symspell": symspell,
        "nlp": nlp,
        "text": ""
    }
    
    pipeline_steps = [
        {"name": "header_footer", "priority": 10, "enabled": True, "func": step_detect_header_footer},
        {"name": "ligature", "priority": 20, "enabled": True, "func": step_ligature_replacement},
        {"name": "regex", "priority": 30, "enabled": True, "func": step_regex_filtering},
        {"name": "exclusion", "priority": 40, "enabled": True, "func": step_exclusion_filtering},
        {"name": "spellcheck", "priority": 50, "enabled": True, "func": step_spellcheck},
        {"name": "stopword_removal", "priority": 55, "enabled": True, "func": step_stopword_removal},  # <-- NEW
#        {"name": "ner", "priority": 60, "enabled": True, "func": step_ner_redaction},
        {"name": "build", "priority": 70, "enabled": True, "func": step_build_paragraph}
    ]

    steps = sorted([s for s in pipeline_steps if s["enabled"]], key=lambda x: x["priority"])

    # Process PDF
    for page_layout in extract_pages(pdf_path):
        for element in page_layout:
            if isinstance(element, LTTextContainer):
                raw_text, font_sizes = collect_text_blocks(element)
                if not raw_text:
                    continue

                context["text"] = raw_text
                for step in steps:
                    context = step["func"](raw_text, font_sizes, context)

    context = finalize_paragraphs(context)
    context = clean_paragraphs(context)

    return context["paragraphs"]

def load_symspell(dictionary_path, max_edit_distance=2, prefix_length=7):
    sym_spell = SymSpell(max_dictionary_edit_distance=max_edit_distance, prefix_length=prefix_length)
    
    if not os.path.exists(dictionary_path):
        raise FileNotFoundError(f"SymSpell dictionary not found at {dictionary_path}")
    
    sym_spell.load_dictionary(dictionary_path, term_index=0, count_index=1)
    return sym_spell
  
def collect_text_blocks(element):
    text = element.get_text().strip()
    font_sizes = []
    for line in element:
        if hasattr(line, "objs"):
            for char in line:
                if isinstance(char, LTChar):
                    font_sizes.append(char.size)
    return text, font_sizes

def step_detect_header_footer(text, font_sizes, context):
    avg_font_size = sum(font_sizes) / len(font_sizes) if font_sizes else 0
    if avg_font_size < context["font_size_threshold"]:
        context["header_footer_texts"].append(text)
    return context

def step_ligature_replacement(text, font_sizes, context):
    context["text"] = replace_private_use_char(text)
    return context

def step_regex_filtering(text, font_sizes, context):
    filtered = remove_patterns_from_text(context["text"], context["regex_patterns"])
    context["text"] = filtered
    return context

def step_exclusion_filtering(text, font_sizes, context):
    tokens = context["text"].split()
    filtered = [t for t in tokens if t.lower() not in context["exclusion_words"]]
    context["text"] = " ".join(filtered)
    return context

def step_spellcheck(text, font_sizes, context):
    tokens = context["text"].split()
    corrected_tokens = []

    for token in tokens:
        # Skip numeric tokens
        if token.isnumeric():
            corrected_tokens.append(token)
            continue

        suggestions = context["symspell"].lookup(token, Verbosity.CLOSEST, max_edit_distance=2)
        if suggestions:
            corrected_tokens.append(suggestions[0].term)
        else:
            corrected_tokens.append(token)

    context["text"] = " ".join(corrected_tokens)
    return context

def step_ner_redaction(text, font_sizes, context):
    doc = context["nlp"](context["text"])
    redacted = [ent.text for ent in doc.ents]
    for name in redacted:
        context["text"] = context["text"].replace(name, "[REDACTED]")
    return context

def step_build_paragraph(text, font_sizes, context):
    final_text = context["text"]
    if len(final_text) < context["min_char_per_paragraph"]:
        return context
    if context["current_paragraph"]:
        context["current_paragraph"] += " " + final_text
    else:
        context["current_paragraph"] = final_text
    if final_text.endswith(('.', '?', '!')):
        context["paragraphs"].append(context["current_paragraph"].strip())
        context["current_paragraph"] = ""
    return context
  
def collect_text_blocks(element):
    """Extract text and font sizes from a PDF layout element."""
    text = element.get_text().strip()
    font_sizes = []

    if hasattr(element, "__iter__"):
        for text_line in element:
            if hasattr(text_line, "objs"):
                for char in text_line:
                    if isinstance(char, LTChar):
                        font_sizes.append(char.size)

    return text, font_sizes

def detect_header_footer(text, font_sizes, context):
    """Collect text blocks suspected to be headers/footers based on font size."""
    avg_font_size = sum(font_sizes) / len(font_sizes) if font_sizes else 0
    if avg_font_size < context["font_size_threshold"]:
        context["header_footer_texts"].append(text)
    return context

def build_paragraph(text, context):
    """Build paragraphs from flowing text."""
    if len(text) < context["min_char_per_paragraph"]:
        return context

    if context["current_paragraph"]:
        context["current_paragraph"] += " " + text
    else:
        context["current_paragraph"] = text

    if text.endswith(('.', '?', '!')):
        context["paragraphs"].append(context["current_paragraph"].strip())
        context["current_paragraph"] = ""

    return context

def step_stopword_removal(text, font_sizes, context):
    """Remove stopwords from the current text using spaCy."""
    doc = context["nlp"](context["text"])
    cleaned_tokens = [token.text for token in doc if not token.is_stop]
    context["text"] = " ".join(cleaned_tokens)
    return context

def finalize_paragraphs(context):
    """Finalize any paragraph left over after processing."""
    if context["current_paragraph"]:
        context["paragraphs"].append(context["current_paragraph"].strip())
        context["current_paragraph"] = ""
    return context

def clean_paragraphs(context):
    """Apply cleaning functions to paragraphs."""
    paragraphs = [clean_paragraph_text(p) for p in context["paragraphs"]]
    paragraphs = remove_common_headers_footers(paragraphs, context["header_footer_texts"])
    if context["split_large_paragraphs"]:
        paragraphs = split_long_paragraphs(paragraphs, context["max_words_per_chunk"])

    context["paragraphs"] = paragraphs
    return context
def replace_private_use_char(text):
    replacements = {
        '\uf001': 'fi', '\uf002': 'fl',
        '\ufb00': 'ff', '\ufb01': 'fi', '\ufb02': 'fl',
        '\ufb03': 'ffi', '\ufb04': 'ffl', '\ufb05': 'ft', '\ufb06': 'st'
    }
    for char, replacement in replacements.items():
        text = text.replace(char, replacement)
    return unicodedata.normalize('NFKC', text)

def remove_patterns_from_text(text, patterns):
    for pattern in patterns:
        text = re.sub(pattern, '', text, flags=re.MULTILINE)
    return text.strip()

def clean_paragraph_text(paragraph):
    paragraph = re.sub(r'-\s+', '', paragraph)  # Remove hyphenation
    paragraph = re.sub(r'\s+', ' ', paragraph)  # Normalize spaces
    return paragraph.strip()

def remove_common_headers_footers(paragraphs, header_footer_texts, threshold=5):
    common_texts = [text for text, count in Counter(header_footer_texts).items() if count >= threshold]
    return [p for p in paragraphs if not any(common in p for common in common_texts)]

def batch_extract_and_cache(root_folder, cache_folder,
                             min_char_per_paragraph=50,
                             font_size_threshold=7,
                             max_words_per_chunk=150,
                             log_file="processing_log.csv"):
    """
    Process all PDFs by decade, cache cleaned paragraphs in JSON files, and log processing results.
    """
    decades = sorted(os.listdir(root_folder))
    processing_log = []

    for decade in tqdm(decades, desc="Processing Decades"):
        decade_path = os.path.join(root_folder, decade)
        if not os.path.isdir(decade_path):
            continue

        decade_cache_path = os.path.join(cache_folder, decade)
        os.makedirs(decade_cache_path, exist_ok=True)

        for filename in os.listdir(decade_path):
            if not filename.lower().endswith(".pdf"):
                continue

            pdf_path = os.path.join(decade_path, filename)
            cache_file = os.path.join(decade_cache_path, filename.replace('.pdf', '.json'))
            doc_name = filename.replace('.pdf', '')

            # Skip if already cached
            if os.path.exists(cache_file):
                processing_log.append({
                    "decade": decade,
                    "document": doc_name,
                    "chunks_extracted": 0,
                    "status": "skipped (already cached)"
                })
                continue

            try:
                paragraphs = extract_clean_paragraphs_from_pdf(
                    pdf_path,
                    min_char_per_paragraph,
                    font_size_threshold,
                    split_large_paragraphs=True,
                    max_words_per_chunk=max_words_per_chunk
                )

                if not paragraphs:
                    processing_log.append({
                        "decade": decade,
                        "document": doc_name,
                        "chunks_extracted": 0,
                        "status": "skipped (empty)"
                    })
                    continue

                chunk_records = [{
                    "text": p,
                    "document": doc_name,
                    "decade": decade,
                    "chunk_id": idx
                } for idx, p in enumerate(paragraphs)]

                with open(cache_file, 'w', encoding='utf-8') as f:
                    json.dump(chunk_records, f, ensure_ascii=False, indent=2)

                processing_log.append({
                    "decade": decade,
                    "document": doc_name,
                    "chunks_extracted": len(chunk_records),
                    "status": "success"
                })

            except Exception as e:
                processing_log.append({
                    "decade": decade,
                    "document": doc_name,
                    "chunks_extracted": 0,
                    "status": f"failed ({str(e)[:50]})"
                })

    # Save processing log at the end
    log_path = os.path.join(cache_folder, log_file)
    pd.DataFrame(processing_log).to_csv(log_path, index=False)
    print(f"✅ Processing log saved to {log_path}")

def split_long_paragraphs(paragraphs, max_words=150):
    """
    Split paragraphs longer than max_words into smaller chunks.
    """
    split_paragraphs = []
    for para in paragraphs:
        words = para.split()
        for start in range(0, len(words), max_words):
            chunk = ' '.join(words[start:start + max_words])
            split_paragraphs.append(chunk)
    return split_paragraphs
def load_or_train_bertopic(docs, model_output_name, embedding_model,
                           n_gram_range=(1, 2), min_topic_size=30, verbose=True):
    """
    Load a BERTopic model if it exists; otherwise, train a new one.
    
    Args:
        docs (list of str): List of text chunks.
        model_output_name (str): Path to save/load the BERTopic model.
        embedding_model (SentenceTransformer): Embedding model instance.
        n_gram_range (tuple): n-gram range for vectorizer (default: (1,2)).
        min_topic_size (int): Minimum topic size for BERTopic clustering.
        verbose (bool): Whether to show progress logs.
    
    Returns:
        topic_model: Fitted BERTopic model.
        topics: List of topic assignments for each document.
        probs: List of probabilities for each document.
    """

    if not docs:
        raise ValueError("🚨 'docs' list is empty. Cannot train or transform BERTopic.")

    if os.path.exists(model_output_name):
        print("✅ Found existing BERTopic model. Loading...")
        topic_model = BERTopic.load(model_output_name)

        # Encode documents manually
        embeddings = embedding_model.encode(
            docs, 
            show_progress_bar=True,
            batch_size=32,
            convert_to_numpy=True
        )

        topics, probs = topic_model.transform(docs, embeddings=embeddings)
        
    else:
        print("🚀 No model found. Training a new BERTopic model...")
        
        topic_model = BERTopic(
            embedding_model=embedding_model,
            n_gram_range=n_gram_range,
            min_topic_size=min_topic_size,
            verbose=verbose
        )

        topics, probs = topic_model.fit_transform(docs)
        topic_model.save(model_output_name)
        print(f"✅ BERTopic model saved to {model_output_name}")

    return topic_model, topics, probs

def load_cached_data(cache_folder):
    """
    Load cached JSON files into lists for BERTopic modeling.
    
    Returns:
    - all_chunks: List of cleaned paragraph strings
    - doc_sources: List of (decade, document, chunk_id) tuples
    """
    all_chunks = []
    doc_sources = []

    # Ensure the decades are sorted properly
    decades = sorted(os.listdir(cache_folder))
    
    for decade in decades:
        decade_path = os.path.join(cache_folder, decade)
        if not os.path.isdir(decade_path):
            continue  # Skip non-folders

        for filename in os.listdir(decade_path):
            if filename.endswith('.json'):
                cache_file = os.path.join(decade_path, filename)
                try:
                    with open(cache_file, 'r', encoding='utf-8') as f:
                        records = json.load(f)

                        # Handle if the JSON is a dict, not a list
                        if isinstance(records, dict):
                            records = [records]

                        for record in records:
                            if not isinstance(record, dict):
                                continue  # Skip malformed record
                            
                            text = record.get("text")
                            decade_val = record.get("decade", decade)
                            document_val = record.get("document", filename.replace('.json', ''))
                            chunk_id = record.get("chunk_id", -1)

                            if text and text.strip():  # Avoid empty text
                                all_chunks.append(text.strip())
                                doc_sources.append((decade_val, document_val, chunk_id))
                except Exception as e:
                    print(f"⚠️ Warning: Could not load {cache_file}. Error: {e}")

    print(f"✅ Loaded {len(all_chunks)} text chunks and {len(doc_sources)} doc_sources.")
    return all_chunks, doc_sources
