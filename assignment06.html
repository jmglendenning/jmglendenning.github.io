# app.R

library(shiny)
library(datasets)
library(ggplot2)
library(dplyr)
library(readr)

# ---------------------------------------------------------
# Load Texas dataset (Part D)
# ---------------------------------------------------------
tx_data <- read_csv(
  "tx_county_voting_data_2020.csv",
  show_col_types = FALSE
)

# ---------------------------------------------------------
# Convert uspop to data frame (FIXES sliderInput error)
# ---------------------------------------------------------
uspop_df <- data.frame(
  Year = as.numeric(time(uspop)),
  Population = as.numeric(uspop)
)

# ---------------------------------------------------------
# UI
# ---------------------------------------------------------
ui <- fluidPage(
  
  # ---- Custom Font Styling ----
  tags$style(HTML("
    body {
      background-color: white;
      color: black;
      font-family: 'Helvetica Neue', Arial, sans-serif;
    }
    h1, h2, h3 {
      font-family: 'Helvetica Neue', Arial, sans-serif;
      color: #000000;
    }
    .shiny-input-container {
      color: #000000;
      font-family: 'Helvetica Neue', Arial, sans-serif;
    }
  ")),
  
  titlePanel("Shiny Assignment — Multi-Dataset Dashboard"),
  
  tabsetPanel(
    
    # ------------------------- TAB 1 -------------------------
    tabPanel("mtcars",
             sidebarLayout(
               sidebarPanel(
                 selectInput("mt_var", "Variable to Plot vs mpg",
                             choices = names(mtcars),
                             selected = "hp")
               ),
               mainPanel(
                 plotOutput("mtcarsPlot")
               )
             )
    ),
    
    # ------------------------- TAB 2 -------------------------
    tabPanel("USArrests",
             sidebarLayout(
               sidebarPanel(
                 selectInput("arrest_var", "Choose a variable",
                             choices = names(USArrests),
                             selected = "Murder")
               ),
               mainPanel(
                 plotOutput("arrestsPlot")
               )
             )
    ),
    
    # ------------------------- TAB 3 -------------------------
    tabPanel("uspop",
             sidebarLayout(
               sidebarPanel(
                 sliderInput("pop_years", "Number of years to show:",
                             min = 5,
                             max = nrow(uspop_df),
                             value = 10)
               ),
               mainPanel(
                 plotOutput("popPlot")
               )
             )
    ),
    
    # ------------------------- TAB 4 -------------------------
    tabPanel("Texas County Voting Data",
             sidebarLayout(
               sidebarPanel(
                 selectInput("tx_var", "Variable to Plot",
                             choices = names(tx_data),
                             selected = names(tx_data)[2])
               ),
               mainPanel(
                 plotOutput("txPlot")
               )
             )
    )
  )
)

# ---------------------------------------------------------
# SERVER
# ---------------------------------------------------------
server <- function(input, output) {
  
  # ---- mtcars plot ----
  output$mtcarsPlot <- renderPlot({
    ggplot(mtcars, aes_string(x = input$mt_var, y = "mpg")) +
      geom_point(color = "#2f3b44") +
      theme_minimal(base_family = "Helvetica Neue") +
      labs(title = paste("mpg vs", input$mt_var),
           x = input$mt_var,
           y = "Miles per Gallon")
  })
  
  # ---- USArrests plot ----
  output$arrestsPlot <- renderPlot({
    ggplot(USArrests, aes_string(x = "UrbanPop", y = input$arrest_var)) +
      geom_point(color = "#536cae") +
      theme_minimal(base_family = "Helvetica Neue") +
      labs(title = paste(input$arrest_var, "vs UrbanPop"),
           x = "Urban Population (%)",
           y = input$arrest_var)
  })
  
  # ---- uspop time series plot ----
  output$popPlot <- renderPlot({
    d <- head(uspop_df, input$pop_years)
    
    ggplot(d, aes(x = Year, y = Population)) +
      geom_line(linewidth = 1.1, color = "#3a5f7d") +
      geom_point(size = 2, color = "#3a5f7d") +
      theme_minimal(base_family = "Helvetica Neue") +
      labs(title = "US Population Over Time",
           x = "Year",
           y = "Population")
  })
  
  # ---- Texas voting plot ----
  output$txPlot <- renderPlot({
    ggplot(tx_data, aes_string(x = "County", y = input$tx_var)) +
      geom_col(fill = "#3a5f7d") +
      theme_minimal(base_family = "Helvetica Neue") +
      theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
      labs(title = paste("Texas Counties —", input$tx_var),
           x = "County",
           y = input$tx_var)
  })
}

# ---------------------------------------------------------
# Run App
# ---------------------------------------------------------
shinyApp(ui = ui, server = server)
