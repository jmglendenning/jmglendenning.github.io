library(ggplot2)

# Reusable theme: sans font, gray grid, small caps titles
theme_bluecaps <- theme_minimal(base_family = "sans") +
  theme(
    plot.title = element_text(face = "bold", color = "#08306B",
                              size = 13, hjust = 0.5, family = "sans"),
    axis.title = element_text(face = "plain", color = "gray20"),
    axis.text  = element_text(color = "gray30"),
    panel.grid.minor = element_blank(),
    panel.grid.major = element_line(color = "gray85"),
    plot.caption = element_text(color = "gray40", size = 9, hjust = 0.5)
  )

library(ggplot2)
library(gridExtra)
library(magick)
library(grid)

formats <- c("pdf", "jpg", "svg", "tiff")

# Helper to read an image and caption it
read_any_image <- function(filename, caption) {
  tryCatch({
    img <- magick::image_read(filename)
    gridExtra::arrangeGrob(
      rasterGrob(as.raster(img), interpolate = TRUE),
      bottom = textGrob(
        toupper(caption),
        gp = gpar(fontsize = 10, col = "gray20", fontface = "bold")
      )
    )
  }, error = function(e) {
    gridExtra::arrangeGrob(
      rectGrob(gp = gpar(fill = "gray95", col = NA)),
      bottom = textGrob(
        paste("Missing:", caption),
        gp = gpar(fontsize = 9, col = "gray50")
      )
    )
  })
}

# Collect labeled images
base_imgs <- mapply(

Comparison of Image File Formats for Chart Export
Format Type Compression Transparency Typical_Use Pros Cons
PDF Vector Lossless Yes Reports, Printing Infinite zoom, small files Not ideal for web
JPG Raster Lossy No Web, Slides Small size, universal Blurry on zoom, lossy
SVG Vector None Yes Web, Logos Scalable, editable Limited app support
TIFF Raster Lossless Yes Publication, Print High quality, lossless Large files
BMP Raster Uncompressed No Legacy, Windows Apps Simple, compatible Outdated, large

Note that BMP was skipped as an export format because Quartz on macOS does not handle this well and BMP is essentially deprecated on all platforms.
